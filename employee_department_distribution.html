import pandas as pd
import matplotlib.pyplot as plt
from io import StringIO

# Load data
csv_data = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,Finance,Africa,93.29,3,4.3
EMP002,Finance,Africa,93.82,1,4.1
EMP003,HR,Europe,78.96,5,4.2
EMP004,HR,North America,62.16,7,4.2
EMP005,Finance,Europe,87.72,10,4.4
"""
df = pd.read_csv(StringIO(csv_data))

# Frequency count for "HR"
hr_count = (df["department"] == "HR").sum()
print(f'Frequency count for the "HR" department: {hr_count}')

# Histogram-like bar chart of departments
dept_counts = df["department"].value_counts().sort_index()
plt.figure()
dept_counts.plot(kind="bar")
plt.xlabel("Department")
plt.ylabel("Frequency")
plt.title("Distribution of Departments")
plt.tight_layout()
plt.show()
